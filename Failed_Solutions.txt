 #include <iostream>
#include <vector>
#include <algorithm>

int trap(std::vector<int>& elevationMap) {

    int size = elevationMap.size();
    int height = 0;
    std::sort(elevationMap.begin(), elevationMap.end());

    int right = size - 1, left = right - 1;

    for(int i = 0; i < size / 2; i++) {

        std::cout<< "right: " << elevationMap[right] << "left: " << elevationMap[left] << ", ";

        if(elevationMap[right] == elevationMap[left]) {
            height += elevationMap[right];
        } else if(elevationMap[right] > elevationMap[left]) {
            while(elevationMap[right] > elevationMap[left]) {
                elevationMap[right] -= 1;
                if(elevationMap[right] == elevationMap[left]) {
                    height += elevationMap[right];
                    break;
                }
            }
        }
        right-=2;
        left-= 2;
    }
    return height;
};

int main() {

    std::vector<int> tc2 = {4,2,0,3,2,5};

    std::cout<< trap(tc2);
